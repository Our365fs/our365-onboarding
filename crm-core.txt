async function doLogin(){
  const email=document.getElementById('loginEmail').value.trim().toLowerCase();
  const pass=document.getElementById('loginPass').value;
  if(!email||!pass){showLoginErr('Enter email and password.');return}
  if(!sb){showLoginErr('Cannot connect to database.');return}
  try{
    const{data,error}=await sb.from('agents').select('*').eq('email',email).single();
    if(error||!data){showLoginErr('Account not found.');return}
    const ok=(data.agent_role==='master_admin'&&pass==='Our365Admin')||(data.role==='admin'&&pass==='Our365Admin')||(pass===data.agent_number);
    if(!ok){showLoginErr('Incorrect password.');return}
    currentUser=data;
    viewAgencyId=data.agency_id||'agency_our365';
    sessionStorage.setItem('crm_user',JSON.stringify(currentUser));
    sessionStorage.setItem('crm_agency',viewAgencyId);
    showApp();
  }catch(e){showLoginErr('Connection error.')}
}
